Loading Javascript for a view

From a Razor view, add the following to the @scripts section

	<script type="text/javascript">
        (function () {
            var View = null;
            
            //Take notice of this sytax
            require(['app/home/UsingRequire'], function (view) {
                View = view;
                _init.call(this);
            });

            function _init() {
                var view = new View({
                        clickable       : ".UsingRequire .clickable",
                        templateDisplay : ".UsingRequire .templateDisplay"
                    },
                    {
                        clickColor: ["#002244", "#69BE28", "#A5ACAF"]
                    }
                );
            }//end init
        })();
